student_scores = {
    'Harry': 88,
    'Ron': 78,
    'Hermione': 95,
    'Draco': 75,
    'Neville': 60
}
student_grades =student_scores
for l in student_scores():
    if l>=91 & l<=100:
        student_grades.key="Outstanding"
    if l>=81 & l<=90
        student_grades.key="Exceeds Expectations"
    if l>=71 & l<=80
        student_grades.key="Acceptable"
    if l<=70
        student_grades.key="Fail"
